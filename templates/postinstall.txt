$def with (proxy,salt_master,uid)
#!/bin/sh

# set up salt master
mkdir /srv/salt
mkdir /srv/pillar

/usr/bin/curl -o /root/firstboot http://$proxy/firstboot

chmod +x /root/firstboot


cat > /etc/salt/minion <<EOF
master: $salt_master 
id: $uid 
EOF

cat > /etc/init.d/firstboot <<EOF
### BEGIN INIT INFO
# Provides:        firstboot
# Required-Start:  $$networking
# Required-Stop:   $$networking
# Default-Start:   2 3 4 5
# Default-Stop:    0 1 6
# Short-Description: A script that runs once
# Description: A script that runs once
### END INIT INFO
cd /root ; /usr/bin/nohup sh -x /root/firstboot > /dev/console &
EOF

chmod +x /etc/init.d/firstboot
# remove salt minion until after upgrade
update-rc.d salt-minion remove
update-rc.d firstboot defaults
